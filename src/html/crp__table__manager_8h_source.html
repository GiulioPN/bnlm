73  // seat a customer at a table proportional to the number of customers seated at a table, less the discount
74  // *new tables are never created by this function!
75  // returns the number of customers already seated at the table (always > 0)
76  template<typename Engine>
77  unsigned share_table(const double discount, Engine& eng) {
78  const double z = customers - discount * num_tables();
79  double r = z * sample_uniform01<double>(eng);
80  const auto floor_count = [&]()->std::pair<unsigned,int> {
81  for (unsigned floor = 0; floor < NumFloors; ++floor) {
82  const auto end = h[floor].end();
83  auto i = h[floor].begin();
84  for (; i != end; ++i) {
85  const double thresh = (i->first - discount) * i->second;
86  if (thresh > r) return std::make_pair(floor, i->first);
87  r -= thresh;
88  }
89  }
90  std::cerr << "Serious error while incrementing: Floors=" << NumFloors
91  << " r=" << r << std::endl;
92  std::abort();
93  }();
94  const unsigned cc = floor_count.second;
95  h[floor_count.first].move(cc, cc + 1);
96  ++customers;
97  return cc;
98  }

100  // randomly sample a customer
101  // *tables may be removed
102  // returns (floor,table delta). Will be (0,0) unless a table is removed
103  template<typename Engine>
104  inline std::pair<unsigned,int> remove_customer(Engine& eng, unsigned* selected_table_postcount) {
105  int r = sample_uniform01<double>(eng) * num_customers();
106  const auto floor_count = [&]()->std::pair<unsigned,int> {
107  for (unsigned floor = 0; floor < NumFloors; ++floor) {
108  const auto end = h[floor].end();
109  auto i = h[floor].begin();
110  for (; i != end; ++i) {
111  // sample randomly, i.e. *don't* discount
112  const int thresh = i->first * i->second;
113  if (thresh > r) return std::make_pair(floor, i->first);
114  r -= thresh;
115  }
116  }
117  std::cerr << "Serious error while decrementing: Floors=" << NumFloors
118  << " r=" << r << std::endl;
119  std::abort();
120  }();
121  --customers;
122  const unsigned tc = floor_count.second;
123  if (selected_table_postcount) *selected_table_postcount = tc - 1;
124  if (tc == 1) { // remove customer from table containing a single customer
125  h[floor_count.first].decrement(1);
126  --tables;
127  return std::make_pair(floor_count.first, -1);
128  } else {
129  h[floor_count.first].move(tc, tc - 1);
130  return std::make_pair(0u, 0);
131  }
132  }
