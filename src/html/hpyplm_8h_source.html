38  template<typename Engine>
39  void increment(unsigned w, const std::vector<unsigned>& context, Engine& eng) {
40  const double bo = backoff.prob(w, context);
41  for (unsigned i = 0; i < N-1; ++i)
42  lookup[i] = context[context.size() - 1 - i];
43  auto it = p.find(lookup);
44  if (it == p.end()) {
45  it = p.insert(make_pair(lookup, crp<unsigned>(0.8,0))).first;
46  tr.insert(&it->second); // add to resampler
47  }
48  if (it->second.increment(w, bo, eng))
49  backoff.increment(w, context, eng);
50  }

52  void decrement(unsigned w, const std::vector<unsigned>& context, Engine& eng) {
53  for (unsigned i = 0; i < N-1; ++i)
54  lookup[i] = context[context.size() - 1 - i];
55  auto it = p.find(lookup);
56  assert(it != p.end());
57  if (it->second.decrement(w, eng))
58  backoff.decrement(w, context, eng);

62  for (unsigned i = 0; i < N-1; ++i)
63  lookup[i] = context[context.size() - 1 - i];
64  auto it = p.find(lookup);
